 17    def __init__(self, db_path: str):
 18        """
 19        Initialize database manager.
 20        
 21        Args:

 24        self.db_path = Path(db_path)

 28    def _init_database(self):
 29        """Initialize database tables if they don't exist."""
 30        with sqlite3.connect(self.db_path) as conn:
 31            cursor = conn.cursor()
 32            
 33            # Conference URLs table
 34            cursor.execute('''
 35                CREATE TABLE IF NOT EXISTS conference_urls (
 36                    id INTEGER PRIMARY KEY AUTOINCREMENT,
 37                    language TEXT NOT NULL,
 38                    url TEXT NOT NULL,
 39                    discovered_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 40                    processed BOOLEAN DEFAULT FALSE,
 41                    UNIQUE(language, url)
 42                )
 43            ''')
 44            
 45            # Talk URLs table
 46            cursor.execute('''
 47                CREATE TABLE IF NOT EXISTS talk_urls (
 48                    id INTEGER PRIMARY KEY AUTOINCREMENT,
 49                    conference_url TEXT NOT NULL,
 50                    talk_url TEXT NOT NULL,
 51                    language TEXT NOT NULL,
 52                    discovered_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 53                    processed BOOLEAN DEFAULT FALSE,
 54                    UNIQUE(conference_url, talk_url)
 55                )
 56            ''')
 57            
 58            # Processing log table
 59            cursor.execute('''
 60                CREATE TABLE IF NOT EXISTS processing_log (
 61                    id INTEGER PRIMARY KEY AUTOINCREMENT,
 62                    operation TEXT NOT NULL,
 63                    language TEXT,
 64                    url TEXT,
 65                    status TEXT NOT NULL,
 66                    message TEXT,
 67                    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
 68                )
 69            ''')
 70            
 71            # Talk metadata table
 72            cursor.execute('''
 73                CREATE TABLE IF NOT EXISTS talk_metadata (
 74                    id INTEGER PRIMARY KEY AUTOINCREMENT,
 75                    url TEXT NOT NULL,
 76                    title TEXT NOT NULL,
 77                    author TEXT NOT NULL,
 78                    calling TEXT,
 79                    note_count INTEGER,
 80                    language TEXT,
 81                    year TEXT,
 82                    conference_session TEXT,
 83                    backup_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 84                    UNIQUE(url)
 85                )
 86            ''')
 87            
 88            conn.commit()
 89            self.logger.info("Database initialized successfully")

 91    def store_conference_urls(self, language: str, urls: List[str]) -> int:
 92        """
 93        Store conference URLs in database.
 94        
 95        Args:
 96            language: Language code (eng/spa)
 97            urls: List of conference URLs
 98            
 99        Returns:
100            Number of new URLs stored
101        """
102        stored_count = 0
103        
104        with sqlite3.connect(self.db_path) as conn:
105            cursor = conn.cursor()
106            
107            for url in urls:
108                try:
109                    cursor.execute('''
110                        INSERT OR IGNORE INTO conference_urls 
111                        (language, url) VALUES (?, ?)
112                    ''', (language, url))
113                    
114                    if cursor.rowcount > 0:
115                        stored_count += 1
116                        
117                except sqlite3.Error as e:
118                    self.logger.error(f"Error storing conference URL {url}: {e}")
119            
120            conn.commit()
121        
122        self.logger.info(f"Stored {stored_count} new conference URLs for {language}")
123        return stored_count

125    def get_conference_urls(self, language: str, unprocessed_only: bool = False) -> List[str]:
126        """
127        Retrieve conference URLs from database.
128        
129        Args:
130            language: Language code (eng/spa)
131            unprocessed_only: If True, only return unprocessed URLs
132            
133        Returns:
134            List of conference URLs
135        """
136        with sqlite3.connect(self.db_path) as conn:
137            cursor = conn.cursor()

141            
142            if unprocessed_only:
143                query += ' AND processed = FALSE'
144            
145            cursor.execute(query, params)
146            urls = [row[0] for row in cursor.fetchall()]

201    def get_processing_stats(self) -> Dict[str, Dict[str, Dict[str, int]]]:
202        """Get processing statistics."""
203        with sqlite3.connect(self.db_path) as conn:
204            cursor = conn.cursor()
205            
206            # Conference stats
207            cursor.execute('''
208                SELECT language, 
209                       COUNT(*) as total, 
210                       SUM(CASE WHEN processed = TRUE THEN 1 ELSE 0 END) as processed
211                FROM conference_urls 
212                GROUP BY language
213            ''')
214            conference_stats = {row[0]: {'total': row[1], 'processed': row[2]} 
215                              for row in cursor.fetchall()}
216            
217            # Talk stats
218            cursor.execute('''
219                SELECT language, 
220                       COUNT(*) as total, 
221                       SUM(CASE WHEN processed = TRUE THEN 1 ELSE 0 END) as processed
222                FROM talk_urls 
223                GROUP BY language
224            ''')
225            talk_stats = {row[0]: {'total': row[1], 'processed': row[2]} 
226                         for row in cursor.fetchall()}
227            
228        return {
229            'conferences': conference_stats,
230            'talks': talk_stats
231        }
