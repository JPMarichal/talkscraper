297    def _extract_author(self, soup: BeautifulSoup) -> Optional[str]:
298        """Extract author name from HTML."""
299        # Try multiple selectors for author
300        author_selectors = [
301            '.byline .author',
302            '.author-name',
303            '.byline',
304            '.author',
305            '[data-testid="author"]',
306            '.study-author',
307            'p.author'
308        ]
309        
310        for selector in author_selectors:
311            element = soup.select_one(selector)
312            if element:
313                author = element.get_text(strip=True)
314                # Clean up author text (remove "By " prefix if present)
315                author = re.sub(r'^(By\s+|Por\s+)', '', author, flags=re.IGNORECASE)
316                if author and len(author) > 2:
317                    return author
318        
319        # Fallback: Look for "By [Name]" pattern in the first few paragraphs
320        paragraphs = soup.find_all('p')[:5]
321        for p in paragraphs:
322            text = p.get_text(strip=True)
323            # Look for "By Name" or "Por Name" pattern
324            match = re.search(r'^(By|Por)\s+([A-Z][a-zA-Z\s\.]+)', text, re.IGNORECASE)
325            if match:
326                author = match.group(2).strip()
327                if len(author) > 2:
328                    return author
329        
330        return None

332    def _extract_calling(self, soup: BeautifulSoup) -> Optional[str]:
333        """Extract speaker's calling/position from HTML."""
334        # Try multiple selectors for calling
335        calling_selectors = [
336            '.byline .calling',
337            '.author-calling',
338            '.calling',
339            '.position',
340            '[data-testid="calling"]',
341            '.study-calling'
342        ]
343        
344        for selector in calling_selectors:
345            element = soup.select_one(selector)
346            if element:
347                calling = element.get_text(strip=True)
348                if calling and len(calling) > 3:
349                    return calling
350        
351        # Sometimes calling is in the same element as author, separated by comma or line break
352        byline_element = soup.select_one('.byline')
353        if byline_element:
354            byline_text = byline_element.get_text(separator='\n', strip=True)
355            lines = [line.strip() for line in byline_text.split('\n') if line.strip()]
356            if len(lines) >= 2:
357                # Second line often contains the calling
358                calling = lines[1]
359                if calling and len(calling) > 3:
360                    return calling
361        
362        # Alternative: Look for calling information in paragraphs near the author
363        paragraphs = soup.find_all('p')[:10]
364        for p in paragraphs:
365            text = p.get_text(strip=True)
366            # Common patterns for callings
367            calling_patterns = [
368                r'(President|Elder|Bishop|Member|Sister|Brother|Apostle)\s+of\s+',
369                r'(First|Second)\s+Counselor',
370                r'Presiding\s+Bishop',
371                r'General\s+(Authority|Officer)',
372                r'Quorum\s+of\s+the\s+Twelve',
373                r'First\s+Presidency'
374            ]
375            
376            for pattern in calling_patterns:
377                if re.search(pattern, text, re.IGNORECASE):
378                    # Extract the calling (first sentence or line)
379                    calling = text.split('.')[0].strip()
380                    if len(calling) > 5 and len(calling) < 100:
381                        return calling
382        
383        return None

388        content_selectors = [

399            if element:
400                content_element = element
401                break
402        
403        if not content_element:
404            # Fallback: try to get all paragraphs
405            content_element = soup
406        
407        # Extract paragraphs while preserving HTML structure
408        paragraphs = content_element.find_all('p')
409        if not paragraphs:
410            return None
411        
412        formatted_paragraphs = []
413        for p in paragraphs:
414            # Skip paragraphs that might be metadata, bylines, etc.
415            p_text = p.get_text(strip=True)
416            if len(p_text) < 20:
417                continue
418            
419            # Format the paragraph while preserving important HTML elements
420            formatted_p = self._format_paragraph_html(p)
421            if formatted_p:
422                formatted_paragraphs.append(formatted_p)
423        
424        if formatted_paragraphs:
425            return '\n'.join(formatted_paragraphs)
426        
427        return None

429    def _format_paragraph_html(self, paragraph) -> str:
430        """Format a single paragraph while preserving important HTML elements."""
431        try:
432            # Create a copy to avoid modifying the original
433            p_copy = BeautifulSoup(str(paragraph), 'html.parser')
434            p_elem = p_copy.find('p')
435            
436            if not p_elem:
437                return ""
438            
439            # Remove all links except note references (preserve text content)
440            for link in p_elem.find_all('a'):
441                href = link.get('href', '')
442                class_attr = link.get('class', [])
443                
444                # Keep note reference links
445                if 'note-ref' in class_attr or '#note' in href:
446                    # Convert to our internal note link format
447                    note_match = re.search(r'#note(\d+)', href)
448                    if note_match:
449                        note_id = note_match.group(1)
450                        link['href'] = f"#note{note_id}"
451                        link['class'] = 'note-link'
452                    continue
453                
454                # For all other links, replace with just the text content
455                link.replace_with(link.get_text())
456            
457            # Get HTML content without escaping
458            content = str(p_elem).replace('&lt;', '<').replace('&gt;', '>').replace('&amp;', '&')
459            
460            # Remove unnecessary attributes using regex
461            content = re.sub(r' data-aid="[^"]*"', '', content)
462            content = re.sub(r' id="[^"]*"', '', content)
463            content = re.sub(r' data-scroll-id="[^"]*"', '', content)
464            
465            # Ensure note superscripts are visible by adding text content
466            # Fix <sup class="marker" data-value="X"></sup> to <sup>X</sup>
467            content = re.sub(r'<sup class="marker" data-value="(\d+)"></sup>', r'<sup>\1</sup>', content)
468            
469            # Clean up whitespace but preserve structure
470            content = re.sub(r'\s+', ' ', content)
471            content = content.strip()
472            
473            return content
474            
475        except Exception as e:
476            # Fallback: return plain text if HTML processing fails
477            return f"<p>{paragraph.get_text(strip=True)}</p>"

479    def _extract_notes_selenium(self, url: str) -> Optional[List[str]]:
480        """
481        Extract notes using Selenium for JavaScript-rendered content.
482        
483        Args:
484            url: Talk URL to extract notes from
485            
486        Returns:
487            List of notes or None if extraction fails
488        """
489        driver = None
490        notes = []
491        
492        try:
493            self.logger.debug(f"Starting Selenium note extraction for: {url}")
494            driver = webdriver.Chrome(options=self.chrome_options)
495            
496            # Load the page
497            driver.get(url)
498            
499            # Wait for page to load - reduced timeout for speed
500            WebDriverWait(driver, 5).until(
501                EC.presence_of_element_located((By.TAG_NAME, "body"))
502            )
503            
504            # Try to activate "Related Content" button
505            self._activate_related_content(driver)
506            
507            # Reduced wait time for faster processing
508            time.sleep(1.5)
509            
510            # Extract notes from li elements with id starting with "note"
511            note_elements = driver.find_elements(By.CSS_SELECTOR, 'li[id^="note"]')
512            self.logger.debug(f"Found {len(note_elements)} note elements")
513            
514            for element in note_elements:
515                note_id = element.get_attribute('id')
516                try:
517                    # Get the innerHTML for complete content
518                    inner_html = driver.execute_script("return arguments[0].innerHTML;", element)
519                    
520                    if inner_html and inner_html.strip():
521                        # Clean the HTML to get text only with proper spacing
522                        soup = BeautifulSoup(inner_html, 'html.parser')
523                        # Use separator to ensure spaces between elements
524                        clean_text = soup.get_text(separator=' ', strip=True)
525                        
526                        # Clean up multiple consecutive spaces but preserve single spaces
527                        clean_text = re.sub(r'\s+', ' ', clean_text).strip()
528                        
529                        if clean_text and len(clean_text) > 5:
530                            notes.append(f"[{note_id}] {clean_text}")
531                            self.logger.debug(f"Extracted note {note_id}: {clean_text[:50]}...")
532                        
533                except Exception as e:
534                    self.logger.warning(f"Error extracting note {note_id}: {e}")
535            
536            self.logger.info(f"Successfully extracted {len(notes)} notes from: {url}")
537            return notes
538            
539        except Exception as e:
540            self.logger.error(f"Error in Selenium note extraction for {url}: {e}")
541            return None
542        finally:
543            if driver:
544                driver.quit()

765        
766        Args:

777            'successful': 0,
778            'failed': 0,

788            try:
